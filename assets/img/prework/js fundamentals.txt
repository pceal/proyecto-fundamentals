function calcularMedia(array) {
  if (array.length === 0) {
    return 0; // Evita la división por cero si el array está vacío
  }
  const suma = array.reduce((acumulador, valor) => acumulador + valor, 0);
  const media = suma / array.length;
  return media;
}
